ALGORITHM Sum_Distinct_Elements
VAR
    set1 : ARRAY_OF INTEGER[4] := [3, 1, 7, 9];
    set2 : ARRAY_OF INTEGER[5] := [2, 4, 1, 9, 3];
    sum : INTEGER := 0;
    i, j : INTEGER;
    found : BOOLEAN;
BEGIN
    // Step 1: Check elements of set1 that are NOT in set2
    FOR i FROM 0 TO set1.length - 1 DO
        found := FALSE;
        FOR j FROM 0 TO set2.length - 1 DO
            IF (set1[i] = set2[j]) THEN
                found := TRUE;
                BREAK; // Stop searching if match is found
            END_IF
        END_FOR
        IF (NOT found) THEN
            sum := sum + set1[i];
        END_IF
    END_FOR

    // Step 2: Check elements of set2 that are NOT in set1
    FOR i FROM 0 TO set2.length - 1 DO
        found := FALSE;
        FOR j FROM 0 TO set1.length - 1 DO
            IF (set2[i] = set1[j]) THEN
                found := TRUE;
                BREAK;
            END_IF
        END_FOR
        IF (NOT found) THEN
            sum := sum + set2[i];
        END_IF
    END_FOR

    Write("The sum of distinct elements is: ", sum);
END

FUNCTION dot_product_func(v1, v2 : ARRAY_OF FLOAT) : FLOAT
VAR
    i : INTEGER;
    res : FLOAT := 0;
BEGIN
    FOR i FROM 0 TO v1.length - 1 DO
        res := res + (v1[i] * v2[i]);
    END_FOR
    RETURN res;
END

ALGORITHM orthogonality_check_v2
VAR
    v1, v2 : ARRAY_OF FLOAT[3];
    i, n : INTEGER;
BEGIN
    Read(n);
    FOR i FROM 1 TO n DO
        Read(v1); Read(v2);
        // We use the function inside the condition directly
        IF (dot_product_func(v1, v2) = 0) THEN
            Write("Vectors are orthogonal");
        ELSE
            Write("Vectors are not orthogonal");
        END_IF
    END_FOR
END
